 61def _encode_string(string:Iterable[str], codes:dict) -> NDArray[np.uint8]:
 62    """Perform substitution of string to values in codes."""
 63    nt_arr = np.zeros((len(string),), dtype=np.uint8)
 64    for i, n in enumerate(string):
 65        try:
 66            nt_arr[i] = codes[n]
 67        except KeyError:
 68            nt_arr[i] = codes['?']
 69    return nt_arr

383
384        # get readID where is_cpg > min_cpg
385        read_ids = cpg_counts.filter(
386            compute.greater_equal(cpg_counts.column('is_cpg_sum'), min_cpg)

392    def filter_noncpg_met_per_read(self, max_noncpg=0):
393        # filter by methylation type

395            self.methylation,

399        # count freq noncpg per read
400        noncpg = self.readID.filter(noncpg_mask)
401        # returns a struct with fields "counts", "values"

552def get_insertion_mask(a:AlignedSegment,):
553    """
554    Returns a list where:
555    - 0 indicates a matched base (CIGAR operation 0),
556    - 1 indicates an inserted base (CIGAR operation 1).
557
558    Args:
559        a (pysam.AlignedSegment): The aligned read.
560
561    Returns:
562        list: A list of 0s and 1s indicating matched and inserted bases.
563    """
564    match_insertion_list = []
565
566    for (operation, length) in a.cigartuples:
567        if operation == 0:  # Match
568            match_insertion_list.extend([0] * length)
569        elif operation == 1:  # Insertion
570            match_insertion_list.extend([1] * length)
571        if not len(match_insertion_list) == a.query_length:
572            s = f"Query length did not match calculation. cigar={a.cigartuples}, readName={a.query_name}"
573            raise ValueError(s)
574        # else: continue or handle other operations based on requirements
575
576    return match_insertion_list
