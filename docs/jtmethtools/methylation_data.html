 51class MethylationDataset:
 52    """Locus and read level data with process tracking for data integrity tracking.
 53
 54    Both `locus_data` and `read_data` have column AlignmentIndex, which links the tables.
 55
 56    Methods that alter data return a copy by default, use inplace=True to mutate in place.
 57    """
 58    def __init__(
 59            self,
 60            locus_data:pd.DataFrame,
 61            read_data:pd.DataFrame,
 62            processes:list[dict]=None,
 63            name:str=None,
 64    ):
 65        self.locus_data = locus_data
 66        self.read_data = read_data
 67        self.processes = [p for p in (processes or []) if p is not None]
 68        self.name = name
 69

 73    #     return self.processes

 75    @classmethod
 76    def from_dir(cls, datdir:Path|str) -> Self:
 77        """Create MethylationDataset from directory of data."""
 78        datdir = Path(datdir)
 79        locdat, readdat, processes = load_methylation_data(datdir)
 80        return cls(
 81            locus_data=locdat,
 82            read_data=readdat,
 83            processes=processes,
 84            name=datdir.name,
 85        )
 86
 87    def write_to_dir(self, outdir:Path|str, create_outdir=True) -> None:
 88        """Write data tables to parquet files, and metadata to a JSON.
 89
 90        `outdir` will be created if it doesn't exist."""
 91        outdir = Path(outdir)
 92        if create_outdir:
 93            outdir.mkdir(parents=True, exist_ok=True)
 94        self.locus_data.to_parquet(outdir/'locus_data.parquet')
 95        self.read_data.to_parquet(outdir/'read_data.parquet')
 96        with open(outdir/'metadata.json', 'w') as f:
 97            json.dump(self.processes, f, indent=4)
 98
 99    def to_dir(self, outdir:Path|str, create_outdir=True) -> None:
100        """Alias for write_to_dir."""
101        self.write_to_dir(outdir, create_outdir)

104    def get_process_record(self, name:str, parameters:dict=None, inplace=False) -> dict:
105        """Record a data processing step in the processes list.
106
107        Set inplace=True to update self.processes list in place."""
108        new_entry = {
109                'name': name,
110                'date_time': str(pd.Timestamp.now()),
111                **parameters,
112            }
113
114        if inplace:
115            self.processes.append(new_entry)
116
117        return new_entry

120    def drop_alignments(self, aln_ids, inplace=False) -> Self|None:
121        """Drop given alignments from read and locus data tables."""
122        locdat = self.locus_data.loc[~self.locus_data.AlignmentIndex.isin(aln_ids)].reset_index(drop=True)
123        reads_init = self.read_data.shape[0]
124        readat = self.read_data.loc[~self.read_data.AlignmentIndex.isin(aln_ids)].reset_index(drop=True)
125        procs = self.get_process_record(
126            name='drop_alignments',
127            parameters={'num_reads_removed': reads_init-readat.shape[0], 'reads_remaining':readat.shape[0]},
128            inplace=inplace
129        )
130
131        if not inplace:
132            return MethylationDataset(
133                locus_data=locdat,
134                read_data=readat,
135                processes=self.processes + [procs],
136            )
137        else:
138            self.locus_data = locdat
139            self.read_data = readat
140            return None

142    def harmonise_tables(self, inplace=False):
143        """Match alignment IDs, removing alignments from either table that don't appear in the other."""
144        loc_aln = set(self.locus_data.AlignmentIndex.unique())
145        read_aln = set(self.read_data.AlignmentIndex.unique())
146        aln_to_keep = loc_aln.intersection(read_aln)
147
148        proc_rec = dict(name='harmonise_tables', parameters={'num_alignments_removed': len(loc_aln.symmetric_difference(read_aln))})
149        if inplace:
150            self.get_process_record(**proc_rec, inplace=True)
151            self.drop_alignments(aln_to_keep, inplace=True)
152            return None
153        else:
154
155            mdat = self.drop_alignments(aln_to_keep)
156            mdat.get_process_record(**proc_rec, inplace=True)
157            return mdat
158
159    # getitem to allow tuple unpacking, e.g. locus, read, meta = dataset
160    def __getitem__(self, index):
161        if index == 0:
162            return self.locus_data
163        elif index == 1:
164            return self.read_data
165        elif index == 2:
166            return self.processes
167        else:
168            raise IndexError("MethylationDataset only has three items: locus_data, read_data, metadata.")

186def process_bam_methylation_data(
187        bamfn:Path|str,
188        first_i:int=0,
189        last_i:int=np.inf,
190        regions:Regions=None,
191        paired_end=True,
192        cannonical_chrm_only=True,
193        include_unmethylated_ch=False,
194        chunk_size=int(1e6),
195        min_mapq=0,
196        #min_phred=0, # this one's easier at the table level
197) -> MethylationDataset:
198
199    logger.info(f"Processing BAM, {bamfn}, {first_i}-{last_i}.")
200
201    # Create mappings for dictionary encodings
202    def mapping_to_pa_dict(mapping:dict[str, int], values:NDArray, ) -> pa.DictionaryArray:
203        """Take a encoded numpy array and the dictionary that decodes
204        it, return the dictionary encoded arrow array."""
205        # get values in the order of the mapping so they'll be the same in the
206        #  final dictionary
207        i2s = {v:k for k, v in mapping.items()}
208        dict_values = [i2s[i] for i in range(max(i2s.keys()) + 1)]
209        dictionary = pa.array(dict_values, type=pa.string())
210
211        # Convert to dict array. Pandas needs 32bit array to convert to
212        indices = pa.array(values, type=pa.int32())
213        dict_array = pa.DictionaryArray.from_arrays(indices, dictionary)
214        return dict_array
215
216    # 1. Chromosome mapping
217    with pysam.AlignmentFile(bamfn) as af:
218        header = af.header.to_dict()
219        chrm_map = {}
220        chrm_i = 0
221        for s in header['SQ']:
222            ref = s['SN']
223            if (not cannonical_chrm_only) or (ref in CANNONICAL_CHRM):
224                chrm_map[ref] = chrm_i
225                chrm_i += 1
226    if len(chrm_map) > 255:
227        chrm_dtype = np.int16
228    else:
229        chrm_dtype = np.int8
230
231
232    # 2. nucleotide mapping
233    nt_map = {n: i for i, n in enumerate('NACGT')}
234
235    # 3. Bismark code mapping
236    bismark_map = {n: i for i, n in enumerate('zZhHxXuU.')}
237
238    dictionary_mappings = {'ReadNucleotide': nt_map,
239                            'BismarkCode': bismark_map,
240                            'Chrm': chrm_map}
241
242    if regions is None:
243        filtering_by_region = False
244    else:
245        filtering_by_region = True
246
247    if cannonical_chrm_only and (not filtering_by_region):
248        def include_alignment(a) -> bool:
249            return a.reference_name in CANNONICAL_CHRM
250    elif cannonical_chrm_only and filtering_by_region:
251        def include_alignment(a) -> bool:
252            return (a.reference_name in CANNONICAL_CHRM) and (len(a.get_hit_regions(regions)) > 0)
253    elif not cannonical_chrm_only and filtering_by_region:
254        def include_alignment(a) -> bool:
255            return len(a.get_hit_regions(regions)) > 0
256    elif not cannonical_chrm_only and not filtering_by_region:
257        def include_alignment(a) -> bool:
258            return True
259    else:
260        raise ValueError("Logic error in include_alignment definition. This should not happen.")
261
262
263    if last_i == np.inf:
264        # count em if we aren't processing a subset of the bam
265        i = 0
266        for i, aln in enumerate(iter_bam_segments(bamfn, paired_end)):
267            pass
268        logger.info(f'Reads in BAM: {i}')
269        max_reads = i+1
270    else:
271        max_reads = last_i - first_i
272
273    # note: types that will be dictionary encoded in the Table need to use unsigned integers
274    #  so that they can be converted to pandas DataFrame later
275    read_data = {
276        'AlignmentIndex': np.zeros(max_reads, dtype=np.uint32),
277        'Start': np.zeros(max_reads, dtype=np.uint32),
278        'End': np.zeros(max_reads, dtype=np.uint32),
279        'MappingQuality': np.zeros(max_reads, dtype=np.uint8),
280        'Chrm': np.zeros(max_reads, dtype=chrm_dtype), # dictionary
281    }
282
283    def get_empty_locus_chunk() -> dict[str, NDArray]:
284        return {
285            'AlignmentIndex': np.zeros(chunk_size, dtype=np.uint32),
286            'ReadNucleotide': np.zeros(chunk_size, dtype=np.int8), # dictionary
287            'PhredScore': np.zeros(chunk_size, dtype=np.uint8),
288            'Chrm': np.zeros(chunk_size, dtype=chrm_dtype),  # dictionary
289            'Position': np.zeros(chunk_size, dtype=np.uint32),
290            'ReadPostion': np.zeros(chunk_size, dtype=np.uint32),
291            'BismarkCode': np.zeros(chunk_size, dtype=np.int8),  # dictionary
292        }
293
294    def get_locus_table_from_chunk(loc_data:dict, locus_cursor:int) -> pa.Table:
295        # save the current locus data chunk
296        # convert arrays to pa.Table
297        writable_data = {}
298        for k, array in loc_data.items():
299            array = array[:locus_cursor]
300            # convert the dictionary types
301            if k in dictionary_mappings.keys():
302                writable_data[k] = mapping_to_pa_dict(dictionary_mappings[k], array)
303            else:
304                writable_data[k] = pa.array(array)
305        loc_table = pa.Table.from_pydict(writable_data, )
306        return loc_table
307
308    locus_data_tables = []
309    current_locus_data = get_empty_locus_chunk()
310    locus_cursor = 0
311    read_cursor = 0
312    logger.info('At start for processing alignments {')
313    log_memory_footprint()
314    for aln_i, aln in enumerate(iter_bam(bamfn, paired_end=paired_end)):
315        if aln_i < first_i:
316            continue
317        if aln_i >= last_i:
318            break
319
320        aln_i += 1 # AlignmentIndex 0 is for empty rows.
321
322        aln:Alignment
323
324        if not include_alignment(aln):
325            continue
326
327        if aln.mapping_quality() < min_mapq:
328            continue
329        if aln.a.is_unmapped:
330            continue
331        metstr = aln.metstr
332        if not metstr:
333            # malformed bismark data can result in empty metstr
334            #   for otherwise valid alignments
335            continue
336
337
338        data_added = False
339        for pos, met in aln.locus_methylation.items():
340            if met == '.':
341                continue
342            if include_unmethylated_ch or met.isupper() or (met.lower() == 'z'):
343                data_added = True
344                # record the locus data
345                current_locus_data['AlignmentIndex'][locus_cursor] = aln_i
346                current_locus_data['ReadNucleotide'][locus_cursor] = nt_map[aln.locus_nucleotide[pos]]
347                current_locus_data['PhredScore'][locus_cursor] = aln.locus_quality[pos]
348                current_locus_data['Chrm'][locus_cursor] = chrm_map[aln.reference_name]
349                current_locus_data['Position'][locus_cursor] = pos
350                current_locus_data['BismarkCode'][locus_cursor] = bismark_map[met]
351                current_locus_data['ReadPostion'][locus_cursor] = pos-aln.reference_start
352                #current_locus_data['IsForward'][locus_cursor] = aln.is_forward
353
354                locus_cursor += 1
355
356                if locus_cursor >= chunk_size:
357                    loc_table = get_locus_table_from_chunk(current_locus_data, locus_cursor)
358                    locus_data_tables.append(loc_table)
359                    current_locus_data = get_empty_locus_chunk()
360                    locus_cursor = 0
361        if data_added:
362            # record the read data
363            read_data['AlignmentIndex'][read_cursor] = aln_i
364            read_data['Start'][read_cursor] = aln.reference_start
365            read_data['End'][read_cursor] = aln.reference_end
366            read_data['MappingQuality'][read_cursor] = aln.mapping_quality()
367            read_data['Chrm'][read_cursor] = chrm_map[aln.reference_name]
368            read_cursor += 1
369
370            if read_cursor >= max_reads:
371                logger.info(f'Reached max reads, stopping processing. {bamfn=}, {max_reads=}.')
372
373    # save the last locus data chunk
374    loc_table = get_locus_table_from_chunk(current_locus_data, locus_cursor)
375    locus_data_tables.append(loc_table)
376    locus_table = pa.concat_tables(locus_data_tables)
377
378    # create the read table
379    read_data_final = {}
380    for k, array in read_data.items():
381        array = array[:read_cursor]
382        if k in dictionary_mappings.keys():
383            # convert the dictionary types
384            read_data_final[k] = mapping_to_pa_dict(dictionary_mappings['Chrm'], array)
385        else:
386            read_data_final[k] = pa.array(array)
387
388    read_table = pa.Table.from_pydict(read_data_final, )
389
390    logger.info('Finished processing alignments.')
391    log_memory_footprint()
392
393    process = {
394        'name': 'process_bam_methylation_data',
395        'bam_file': str(Path(bamfn).resolve()), # full path
396        'regions_file': str(regions) if regions is not None else None,
397        'date_time': str(pd.Timestamp.now()),
398        'parameters': {
399            'first_i': first_i,
400            'last_i': last_i,
401            'paired_end': paired_end,
402            'cannonical_chrm_only': cannonical_chrm_only,
403            'include_unmethylated_ch': include_unmethylated_ch,
404            'chunk_size': chunk_size,
405            'min_mapq': min_mapq,
406        }
407    }
408    locus_data = table2df(locus_table)
409    read_data = table2df(read_table)
410    processes = [process]
411
412    locus_data = locus_data.loc[locus_data.AlignmentIndex != 0].reset_index(drop=True)
413    read_data = read_data.loc[read_data.AlignmentIndex != 0].reset_index(drop=True)
414
415    locus_data.loc[:, 'MetCpG'] = locus_data.BismarkCode == 'Z'
416
417    return MethylationDataset(
418        locus_data=locus_data, read_data=read_data,
419        processes=processes
420    )

450def sample_reads(
451        data:MethylationDataset,
452        n_reads:int,
453        with_replacement=False,
454        seed:int=None,
455) -> MethylationDataset:
456    """Sample reads from a methylation dataset."""
457    from copy import copy
458    locus_table, read_table, metadata = data.locus_data.copy(), data.read_data.copy(), copy(data.metadata)
459
460    for t in locus_table, read_table:
461        t.drop('__index_level_0__', axis=1, inplace=True, errors='ignore')
462
463    read_table.set_index('AlignmentIndex', inplace=True, drop=False)
464    locus_table = locus_table.loc[locus_table.AlignmentIndex.isin(read_table.AlignmentIndex)]
465
466    if seed is not None:
467        np.random.seed(seed)
468
469    sampled_alnidx = np.random.choice(
470        read_table.AlignmentIndex, size=n_reads, replace=with_replacement,
471    )
472
473    read_groups = locus_table.groupby('AlignmentIndex').groups
474
475    resamp_read_table = read_table.loc[sampled_alnidx].reset_index(drop=True)
476    resamp_read_table.loc[:, 'OldAlnIdx'] = resamp_read_table.AlignmentIndex
477    resamp_read_table.loc[:, 'AlignmentIndex'] = resamp_read_table.index
478
479    loc_rows = []
480    new_loc_alnIdx = []
481
482    for newidx, row in resamp_read_table.iterrows():
483        loc_rows_idx = read_groups[row.OldAlnIdx]
484        loc_rows.extend(loc_rows_idx)
485        new_loc_alnIdx.extend([newidx] * len(loc_rows_idx))
486
487    resamp_locus_table = locus_table.loc[loc_rows]
488    resamp_locus_table.loc[:, 'AlignmentIndex'] = new_loc_alnIdx
489
490    resamp_read_table.drop('OldAlnIdx', axis=1, inplace=True)
491
492    metadata.get('processes', []).append(
493        f'sample_reads: sampled {n_reads} with_replacement={with_replacement}'
494    )
495
496    methylation_data = MethylationDataset(
497        locus_data=resamp_locus_table,
498        read_data=resamp_read_table,
499        processes=metadata
500    )
501
502    return methylation_data

506def synthetic_sample(
507        inputs:Iterable[tuple[str|Path, float, bool]],
508        target_reads:int,
509) -> MethylationDataset:
510    """Create a synthetic methylation dataset by combining multiple datasets.
511
512    Inputs is a list of tuples of (data_dir, proportion, with_replacement).
513
514    Total reads might end up slightly different due to rounding."""
515
516    logger.info(f"Creating synthetic sample with target reads: {target_reads}")
517
518    locus_tables = []
519    read_tables = []
520    total_reads = 0
521    metadata = [{'Name':'create_synthetic_sample', 'Composition':', '.join([f'{p} from {d}' for d, p, _ in inputs])}]
522
523
524    # normalise the proportions
525    total_proportion = sum([p for _, p, _ in inputs])
526    inputs = [(d, p / total_proportion, w) for d, p, w in inputs]
527
528    # log proportions
529    logger.info("Creating synthetic sample with the following inputs:\n" +
530                '\n'.join([f"  - {p:.1%} from {d} (with_replacement={w})" for d, p, w in inputs]))
531
532    for datdir, proportion, with_replacement in inputs:
533        data = MethylationDataset.from_dir(datdir)
534        n_reads = int(round(target_reads * proportion, 0))
535        sampled_data = sample_reads(data, n_reads, with_replacement=with_replacement)
536
537        # adjust AlignmentIndex in locus data
538        read_table = sampled_data.read_data
539        locus_table = sampled_data.locus_data
540
541        read_table.set_index('AlignmentIndex', inplace=True, drop=False)
542        locus_table = locus_table.loc[locus_table.AlignmentIndex.isin(read_table.AlignmentIndex)]
543
544        alnidx_map = {
545            old_idx: new_idx for new_idx, old_idx
546            in enumerate(read_table.AlignmentIndex, start=total_reads)
547        }
548
549        locus_table.loc[:, 'AlignmentIndex'] = locus_table.AlignmentIndex.map(alnidx_map)
550        read_table.loc[:, 'AlignmentIndex'] = read_table.AlignmentIndex.map(alnidx_map)
551
552        locus_tables.append(locus_table)
553        read_tables.append(read_table)
554
555        total_reads += len(read_table)
556
557    combined_locus_table = pd.concat(locus_tables, ignore_index=True)
558    combined_read_table = pd.concat(read_tables, ignore_index=True)
559
560    synthetic_data = MethylationDataset(
561        locus_data=combined_locus_table,
562        read_data=combined_read_table,
563        processes=metadata
564    )
565
566    logger.info(f'Finished creating synthetic sample. Actual total reads: {total_reads}')
567
568    return synthetic_data

704def write_coverage(locus_data:pd.DataFrame, outfile:str|Path) -> None:
705    """Write a coverage file from locus data."""
706    outfile = Path(outfile)
707    if not locus_data.IsCpG.all():
708        locus_data = locus_data.loc[locus_data.IsCpG]
709
710    if "Chrm" not in locus_data.columns:
711        locus_data = locus_data.reset_index()
712
713    #ld = locus_data.head(10000)
714    ld = locus_data
715    g = ld.groupby(['Chrm', 'Position'])
716    m = g.MetCpG.sum()
717    n = g.MetCpG.count()
718
719    cov = m.reset_index()
720    cov.loc[:, 'Unmet'] = (n - m).values
721    cov.insert(2, 'Perc', (m / n).apply(lambda x: int(round(x * 100))).values)
722    cov.insert(1, 'Pos2', cov.Position)
723
724    # create dir
725    outfile.parent.mkdir(parents=True, exist_ok=True)
726
727    cov.to_csv(
728        outfile,
729        sep='\t', header=None, index=False,
730    )

736        description="Create a synthetic methylation dataset by combining multiple datasets. "
737                    "Example "

706    """Write a coverage file from locus data."""
707    outfile = Path(outfile)
708    if not locus_data.IsCpG.all():
709        locus_data = locus_data.loc[locus_data.IsCpG]
710
711    if "Chrm" not in locus_data.columns:
712        locus_data = locus_data.reset_index()
713

720    cov = m.reset_index()
721    cov.loc[:, 'Unmet'] = (n - m).values
722    cov.insert(2, 'Perc', (m / n).apply(lambda x: int(round(x * 100))).values)
723    cov.insert(1, 'Pos2', cov.Position)
724
