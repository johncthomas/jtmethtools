 49def methylation_by_position(
 50        fn:Path, length=100, stopper=-1, paired_end=True
 51) -> MethFreqResult:
 52    """Calculate the fraction of methylation by position in the read, relative
 53    to the adapter.
 54
 55    Parameters:
 56        fn : Path to the BAM file.
 57        length : Number of bases to consider from the start of the read.
 58        stopper : Stop after this many alignments, -1 means no limit.
 59        paired_end: Set to False if single-ended, to reduce memory usage.
 60    """
 61    logger.info(f"Calculating methylation by position for {fn} with length {length} and stopper {stopper}")
 62    bam_iterer = jtm.alignments.iter_bam_segments(fn, paired_end=paired_end)
 63
 64    mcpg_arr = np.zeros(shape=length, dtype=np.uint32)
 65    cpg_arr = np.zeros(shape=length, dtype=np.uint32)
 66    mch_arr = np.zeros(shape=length, dtype=np.uint32)
 67    ch_arr = np.zeros(shape=length, dtype=np.uint32)
 68
 69    ch_set = {'x', 'h', 'u', 'X', 'H', 'U'}
 70
 71    start = datetime.now()
 72    for aln_i, (a1, a2) in enumerate(bam_iterer):
 73        if a1.is_unmapped or (paired_end and (not a1.is_proper_pair)):
 74            continue
 75        for a in (a1, a2):
 76            if a is None:
 77                continue
 78            # get the methylation string
 79            m = jtm.alignments.get_bismark_met_str(a)
 80            # reverse the read if it's not forward - metstr
 81            #   are always in forward orientation, so this puts the
 82            #   end next to the adapter
 83            if not a.is_forward:
 84                m = ''.join(m[::-1])
 85            for i, met in enumerate(m):
 86                if i > length - 1:
 87                    break
 88                # skip the most common one
 89                if met == '.':
 90                    continue
 91                if met.lower() == 'z':
 92                    cpg_arr[i] += 1
 93                    if met == 'Z':
 94                        mcpg_arr[i] += 1
 95                elif met in ch_set:
 96                    ch_arr[i] += 1
 97                    if met.isupper():
 98                        mch_arr[i] += 1
 99
100        if aln_i == stopper:
101            break
102    end = datetime.now()
103    logger.info(f'Processed {aln_i+1} alignments in {end - start}.')
104
105    return MethFreqResult(
106        cpg_methylated=mcpg_arr,
107        cpg_total=cpg_arr,
108        ch_methylated=mch_arr,
109        ch_total=ch_arr,
110        length=length
111    )

205def iter_filtered_se_v2(
206        bamfn: Path,
207        regions: Regions,
208        counter: Counter,
209        min_mapq: int = 0,
210        min_ncpg: int = 0,
211) -> Iterator[pysam.AlignedSegment]:
212    bam_iterer = jtm.alignments.iter_bam_segments(
213        bamfn, paired_end=False
214    )
215
216
217    for alns in bam_iterer:
218        for aln in alns:
219
220            if aln is None:
221                continue
222
223            counter['TotalReads'] += 1
224
225            if aln.is_unmapped:
226                counter['UnmappedReads'] += 1
227                continue
228
229            if not alignment_overlaps_region(aln, regions):
230                counter['MissesRegions'] += 1
231                continue
232
233            counter['AlignedHits'] += 1
234
235            metstr = jtm.alignments.get_bismark_met_str(aln)
236            low_metstr = metstr.lower()
237
238            if min_mapq and aln.mapping_quality < min_mapq:
239                counter['LowMapQ'] += 1
240                continue
241
242            if min_ncpg and low_metstr.count('z') < min_ncpg:
243                counter['LowCpGs'] += 1
244                continue
245
246            yield aln

262def met_stats_of_regions_v1(
263        bamfn: Path,
264        regions: Regions,
265        min_mapq: int = 0,
266        min_ncpg: int = 0,
267        get_loci_methylation=True
268) -> tuple[Counter, dict[Tuple[str, int], list[int]]]:
269    """Calculate methylation statistics of a filtered BAM file.
270
271    Number/proportion of aligned reads, and number of methylated CpG and CpH.
272
273    Probably makes sense to treat everything as single ended.
274
275    """
276    results = Counter()
277    if get_loci_methylation:
278        methylation_counts:dict[Tuple[str, int], list[int]] = collections.defaultdict(lambda : [0, 0]) # (methylated, total)
279    else:
280        methylation_counts = None
281
282    for aln in iter_filtered_se_v2(
283            bamfn, counter=results, regions=regions, min_mapq=min_mapq, min_ncpg=min_ncpg
284    ):
285        metstr = jtm.alignments.get_bismark_met_str(aln)
286
287
288        for q_pos, r_pos in aln.get_aligned_pairs(matches_only=True):
289
290            met = metstr[q_pos]
291            low_met = met.lower()
292
293            if low_met == 'z':
294                results['TotalCpG'] += 1
295                if met == 'Z':
296                    results['MethylatedCpG'] += 1
297
298
299                vals = methylation_counts[(aln.reference_name, r_pos)]
300                if (met == 'Z'):
301                    vals[0] += 1
302                else:
303                    vals[1] += 1
304
305            elif low_met in {'x', 'h', 'u'}:
306                results['TotalCpH'] += 1
307                if met.isupper():
308                    results['MethylatedCpH'] += 1
309    return results, methylation_counts

398def methylation_beta_density(
399        betas:pd.Series,
400        adjust: float = 1.0,
401        grid_n: int = 2048,
402) -> Tuple[np.ndarray, np.ndarray]:
403    # Clean & clip
404    vals = betas.dropna().to_numpy()
405    vals = vals[(vals >= 0) & (vals <= 1)]
406    if vals.size == 0:
407        return np.array([]), np.array([])
408
409    # Kernel density estimate
410    kde = stats.gaussian_kde(vals, bw_method='scott')
411    kde.set_bandwidth(kde.factor * adjust)
412
413    x = np.linspace(0, 1, grid_n)
414    y = kde(x)
415    return x, y

418def lower_upper_mode(
419        x:np.ndarray, y:np.ndarray,
420        right_threshold: float = 0.6,
421) -> Tuple[float, float]:
422    """
423    Returns (left_mode, right_mode) for a 0-1 bounded series.
424    args:
425        adjust: bandwidth multiplier (≈ R’s `adjust`)
426        grid_n: points in the KDE grid
427        right_threshold: x-value above which to look for the “upper” mode
428        do_plot: if True, make a plot
429    """
430
431    # Local maxima (vectorised & fast)
432    peaks, _ = signal.find_peaks(y)
433    if peaks.size == 0:
434        logger.warning("No modes found.")
435        return 0., 0.
436
437    mode_x = x[peaks]
438    mode_y = y[peaks]
439
440    left_mode = mode_x[np.argmin(mode_x)]  # closest to 0
441    right_candidates = mode_x[mode_x > right_threshold]  # > 0.6 like the R code
442    right_mode = (
443        right_candidates[np.argmax(mode_y[mode_x > right_threshold])]
444        if right_candidates.size else None
445    )
446
447    return left_mode, right_mode

566                                             low_mode, hi_mode)
567                plt.title(f"{sample_name}\n{args.regions.stem}")
568
569                plt.savefig(
570                    out_dir/'beta_plots'/f"{sample_name}.beta-distribution.png",
571                    bbox_inches='tight', dpi=150
572                )
573                plt.close()
574        else:
575            logger.debug("Not getting modes.")
576
577    # write res as a JSON
578    out_json = out_dir / f"{sample_name}.methylation_stats.json"
579    with open(out_json, 'w') as f:
580        json.dump(res, f, indent=4)
581
582    logger.info(f"Results written to {args.out_dir}")

 51        fn:Path, length=100, stopper=-1, paired_end=True
 52) -> MethFreqResult:
 53    """Calculate the fraction of methylation by position in the read, relative
 54    to the adapter.
 55
 56    Parameters:
 57        fn : Path to the BAM file.
 58        length : Number of bases to consider from the start of the read.
 59        stopper : Stop after this many alignments, -1 means no limit.
 60        paired_end: Set to False if single-ended, to reduce memory usage.
 61    """
 62    logger.info(f"Calculating methylation by position for {fn} with length {length} and stopper {stopper}")
 63    bam_iterer = jtm.alignments.iter_bam_segments(fn, paired_end=paired_end)
 64
 65    mcpg_arr = np.zeros(shape=length, dtype=np.uint32)
 66    cpg_arr = np.zeros(shape=length, dtype=np.uint32)
 67    mch_arr = np.zeros(shape=length, dtype=np.uint32)
 68    ch_arr = np.zeros(shape=length, dtype=np.uint32)
 69
 70    ch_set = {'x', 'h', 'u', 'X', 'H', 'U'}
 71
 72    start = datetime.now()
 73    for aln_i, (a1, a2) in enumerate(bam_iterer):
 74        if a1.is_unmapped or (paired_end and (not a1.is_proper_pair)):
 75            continue
 76        for a in (a1, a2):
 77            if a is None:
 78                continue
 79            # get the methylation string
 80            m = jtm.alignments.get_bismark_met_str(a)
 81            # reverse the read if it's not forward - metstr
 82            #   are always in forward orientation, so this puts the
 83            #   end next to the adapter
 84            if not a.is_forward:
 85                m = ''.join(m[::-1])
 86            for i, met in enumerate(m):
 87                if i > length - 1:
 88                    break
 89                # skip the most common one
 90                if met == '.':
 91                    continue
 92                if met.lower() == 'z':
 93                    cpg_arr[i] += 1
 94                    if met == 'Z':
 95                        mcpg_arr[i] += 1
 96                elif met in ch_set:
 97                    ch_arr[i] += 1
 98                    if met.isupper():
 99                        mch_arr[i] += 1
100
101        if aln_i == stopper:
102            break
103    end = datetime.now()
104    logger.info(f'Processed {aln_i+1} alignments in {end - start}.')
105
106    return MethFreqResult(
107        cpg_methylated=mcpg_arr,
108        cpg_total=cpg_arr,
109        ch_methylated=mch_arr,
110        ch_total=ch_arr,
111        length=length
112    )
