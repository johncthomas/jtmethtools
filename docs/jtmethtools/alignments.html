 34def alignment_overlaps_region(
 35        alignment: AlignedSegment,
 36        regions: Regions) -> bool | str:
 37    """Check if an alignment overlaps a region. Returns False if not,
 38    and the name of the first overlapping region if it does.
 39
 40    samtools is much faster if you just want to filter a large BAM."""
 41    ref = alignment.reference_name
 42
 43    try:
 44        regStart, regEnd = regions.starts_ends_of_chrm(ref)
 45    except KeyError:
 46        logger.trace('Reference contig not found: ' + str(alignment.reference_name))
 47        return False
 48
 49    m = (
 50            (alignment.reference_start < regEnd)
 51            & (alignment.reference_end > regStart)
 52    )
 53
 54    isoverlap = np.any(m)
 55
 56    if isoverlap:
 57        reg_name = regions.names[ref][m][0]
 58        logger.trace(reg_name)
 59        return reg_name #its fine
 60    return False

 63def get_bismark_met_str(a: AlignedSegment) -> str|None:
 64    """Return the methylation string for a Bismark alignment, or None if it can't be found."""
 65    # this works in every case I've seen...
 66    tags = a.get_tags()
 67    tag = None
 68    if len(tags) > 2:
 69        tag, met = a.get_tags()[2]
 70    try:
 71        assert tag == 'XM'
 72    # but it's not guaranteed that the tag position will never change
 73    #   so this should work whever it is.
 74    except AssertionError:
 75        for t, m in tags:
 76            if t == 'XM':
 77                tag, met = t, m
 78                break
 79        if tag != 'XM':
 80            logger.warning(f"Can't find XM tag in alignment {a.query_name}, {tags=}")
 81            met = None
 82
 83    return met

 85def write_bam_from_pysam(
 86        out_fn:str,
 87        alignments:Collection[pysam.AlignedSegment],
 88        header_file:str=None,
 89        header:AlignmentHeader=None,
 90):
 91    """Convenience function for writing pysam.AlignedSegments.
 92    Supply a SAM/BAM file name to
 93    use for the header, or supply the header directly. Either
 94    `header_file` or `header`, not both.
 95
 96    Output format (BAM or SAM) determined from out_fn.
 97    """
 98    if (
 99           ( (header_file is None) and (header is None) )
100        or ( (header_file is not None) and (header is not None) )
101    ):
102        raise ValueError("One of either `header_file` or `header` must be supplied.")
103
104    if header_file is not None:
105        header = AlignmentFile(header_file).header
106
107    if str(out_fn).lower().endswith('.bam'):
108        mode = 'wb'
109    elif str(out_fn).lower().endswith('.sam'):
110        mode = 'w'
111    else:
112        raise ValueError(f"Can't determine output file type from file name {out_fn}. "
113                         f"Needs to end with .sam or .bam.")
114    with AlignmentFile(out_fn, mode=mode, header=header) as out:
115        for a in alignments:
116            if a is not None:
117                out.write(a)

120def flag_to_text(flag):
121    """
122    Convert the integer flag from pysam AlignedSegment to a human-readable text representation.
123    """
124    # Flag definitions according to the SAM format specification
125    flag_descriptions = {
126        0x1: "Paired read",
127        0x2: "Properly paired",
128        0x4: "Unmapped read",
129        0x8: "Mate unmapped",
130        0x10: "Read reverse strand",
131        0x20: "Mate reverse strand",
132        0x40: "First in pair",
133        0x80: "Second in pair",
134        0x100: "Not primary alignment",
135        0x200: "Read fails platform/vendor quality checks",
136        0x400: "PCR or optical duplicate",
137        0x800: "Supplementary alignment"
138    }
139
140    result = []
141
142    # Check each flag bit and append the description if the bit is set
143    for flag_value, description in flag_descriptions.items():
144        if flag & flag_value:
145            result.append(description)
146
147    return ", ".join(result) if result else "None"

224    """Wraps one or two pysam AlignedSegments, with methods for getting the
225    methylation state at each reference locus.
226

290    @cached_property
291    def _locus_values(self, ) \
292            -> LocusValues:
293        """Get object with attributes "qualities", "nucleotides", and "methylations"
294        mapping each reference locus to the value, reference_locus -> value.
295        """
296
297        empty_return =  LocusValues({}, {}, {}, is_good=False)
298        if not self._has_metstr():
299            logger.warning(
300                f"Can't determine metstr for alignment of {self.a.query_name}, skipping."
301            )
302            count_alignment_error()
303            return empty_return
304
305        for segment in (self.a, self.a2):
306            if segment is None:
307                continue
308
309            align_len = sum([x[1] for x in segment.cigartuples if x[0] in {0, 1, 7, 8}])
310            metstr_len = len(get_bismark_met_str(segment))
311            phred_len = len(segment.query_qualities)
312            nt_len = len(segment.query_sequence)
313            if not (align_len == metstr_len == phred_len == nt_len):
314                logger.warning(
315                    f"Length mismatch of methylation string for alignment of {self.a.query_name}, skipping.\n"
316                    f"({align_len=}, {metstr_len=}, {phred_len=}, {nt_len=} {segment.cigartuples=}, {self.filename=})"
317                )
318                count_alignment_error()
319                return empty_return
320
321        if self.a2 is None:
322            phreds, methylations, nucleotides = {}, {}, {}
323            for q_pos, r_pos in self.a.get_aligned_pairs(matches_only=True):
324                phreds[r_pos] = self.a.query_qualities[q_pos]
325                nucleotides[r_pos] = self.a.query_sequence[q_pos]
326                methylations[r_pos] = self._get_met_str(self.a)[q_pos]
327        else: # it's a paired alignment
328            if self.use_quality_profile:
329                from jtmethtools._quality_profiles import  quality_profile_match_41, quality_profile_mismatch_41
330            else:
331                quality_profile_match_41, quality_profile_mismatch_41 = None, None
332
333            # .aligned_pairs is reference position->read locus
334            a1_loc_pos, a2_loc_pos = [
335                {r: q for (q, r) in ap if (q is not None) and (r is not None)}
336                for ap in (self.a.get_aligned_pairs(), self.a2.get_aligned_pairs())
337            ]
338            # a1_loc_pos, a2_loc_pos = [
339            #     dict(aln.get_aligned_pairs(matches_only=True))
340            #     for aln in (self.a, self.a2)
341            # ]
342
343            a1_loc = set(a1_loc_pos.keys())
344            a2_loc = set(a2_loc_pos.keys())
345            shared_loc = a1_loc.intersection(a2_loc)
346            a1_only = a1_loc.difference(a2_loc)
347            a2_only = a2_loc.difference(a1_loc)
348
349            phreds = {}
350            nucleotides = {}
351            methylations = {}
352
353            a1_metstr = get_bismark_met_str(self.a)
354            a2_metstr = get_bismark_met_str(self.a2)
355
356            try:
357                # get the values where the position only exists in one of the mates
358                for only_loc, loc_pos, a, metstr in (
359                        (a1_only, a1_loc_pos, self.a, a1_metstr),
360                        (a2_only, a2_loc_pos, self.a2, a2_metstr)
361                ):
362                    for l in only_loc:
363                        p = loc_pos[l]
364                        phreds[l] = a.query_qualities[p]
365                        nucleotides[l] = a.query_sequence[p]
366                        methylations[l] = metstr[p]
367            except:
368                print('BAM = ', self.filename, ' | Read = ', self.a.query_name)
369                raise
370
371            if (not self.prefer_R1) or self.use_quality_profile:
372                for l in shared_loc:
373                    a1_pos = a1_loc_pos[l]
374                    a2_pos = a2_loc_pos[l]
375
376                    a1_nt = self.a.query_sequence[a1_pos]
377                    a2_nt = self.a2.query_sequence[a2_pos]
378
379                    a1_phred = self.a.query_qualities[a1_pos]
380                    a2_phred = self.a2.query_qualities[a2_pos]
381
382                    a1_met = a1_metstr[a1_pos]
383                    a2_met = a2_metstr[a2_pos]
384
385                    # Keep the nucleotide with the highest quality
386                    # (in the case that a1 and a2 have different nucleotides and
387                    #   the phred is the same, we'll keep the a1 NT)
388                    if a1_phred >= a2_phred:
389                        nt = a1_nt
390                        met = a1_met
391                    else:
392                        nt = a2_nt
393                        met = a2_met
394
395                    if self.use_quality_profile:
396                        if a1_nt == a2_nt:
397                            phred = quality_profile_match_41[a1_phred][a2_phred]
398                        else:
399                            phred = quality_profile_mismatch_41[a1_phred][a2_phred]
400                    else:
401                        phred = max((a1_phred, a2_phred))
402
403                    phreds[l] = phred
404                    nucleotides[l] = nt
405                    methylations[l] = met
406            else:
407                # use R1
408                for l in shared_loc:
409                    a1, loc_pos, metstr = (
410                        (self.a, a1_loc_pos, a1_metstr)
411                        if self.a.is_read1
412                        else (self.a2, a2_loc_pos, a2_metstr)
413                    )
414                    a1_pos = loc_pos[l]
415                    phreds[l] = a1.query_qualities[a1_pos]
416                    nucleotides[l] = a1.query_sequence[a1_pos]
417                    methylations[l] = a1_metstr[a1_pos]
418
419        sorted_phreds, sorted_nucleotides, sorted_methylations = (
420            dict(sorted(d.items()))
421            for d in (phreds, nucleotides, methylations)
422        )
423
424        return LocusValues(qualities=sorted_phreds, nucleotides=sorted_nucleotides, methylations=sorted_methylations)

516def _iter_bam_pe_qname_sorted(
517        bam:str|Path|AlignmentFile,
518) -> Iterable[Tuple[AlignedSegment, AlignedSegment|None]]:
519    """Iterate over a paired-end bam file, yielding pairs of alignments.
520    Where a read is unpaired, yield (alignment, None).
521    """
522
523    bam = _load_bam(bam)
524
525    if not bam.header.get('HD', {}).get('SO', 'Unknown') == 'queryname':
526        raise RuntimeError(f'BAM file must be sorted by queryname')
527
528    aln_prev: pysam.AlignedSegment | None = None
529    for i, aln_current in enumerate(bam):
530        logger.debug(f'Alignment #{i}')
531
532
533        if aln_prev is None:
534            aln_prev = aln_current
535            continue
536        elif aln_current.query_name == aln_prev.query_name:
537            yield aln_current, aln_prev
538            aln_prev = None
539        else:
540            yield aln_prev, None
541            aln_prev = aln_current
542
543    if aln_prev is not None:
544        yield aln_prev, None

547def _iter_pe_bam_unsorted(
548        bam:AlignmentFile
549) -> Iterable[Tuple[AlignedSegment, AlignedSegment | None]]:
550    alignment_buffer = {}
551    for a in bam:
552        qn = a.query_name
553        if qn not in alignment_buffer:
554            alignment_buffer[qn] = a
555        else:
556            a1, a2 = a, alignment_buffer[qn]
557            del alignment_buffer[qn]
558            yield a1, a2
559    if alignment_buffer:
560        for a in alignment_buffer.values():
561            yield a, None

581def iter_bam_segments(
582        bam: str | Path | AlignmentFile,
583        paired_end: bool = True,
584) -> Iterable[Tuple[AlignedSegment, AlignedSegment|None]]:
585    """Iterate over a bam file, yielding pairs of alignments, or
586    (segment, None) when it's unpaired.
587
588    Supports SAM and BAM files.
589    """
590    #if check_pairedness, raises exception if bam is actually single-ended"""
591
592    bam = _load_bam(bam)
593    sorting_method = bam.header.get('HD', {}).get('SO', 'Unknown')
594    logger.info(f'{sorting_method=}')
595
596    if paired_end:
597        if (sorting_method == 'queryname'):
598            bamiter = _iter_bam_pe_qname_sorted(bam )
599        elif (sorting_method == 'coordinate'):
600            bamiter = _iter_pe_bam_unsorted(bam)
601        else:
602            raise RuntimeError(
603                "Paired end BAM not sorted by queryname or coordinate. "
604                "Sort it (with -n preferably, but by coordinate is fine), "
605                "or use the --single-ended option.\n"
606            )
607    else:
608        bamiter = _iter_bam_se(bam, )
609    for aln in bamiter:
610        yield aln
611    return None

302            )

305

307            if segment is None:
308                continue
309
310            align_len = sum([x[1] for x in segment.cigartuples if x[0] in {0, 1, 7, 8}])
311            metstr_len = len(get_bismark_met_str(segment))
312            phred_len = len(segment.query_qualities)

314            if not (align_len == metstr_len == phred_len == nt_len):

317                    f"({align_len=}, {metstr_len=}, {phred_len=}, {nt_len=} {segment.cigartuples=}, {self.filename=})"
318                )
319                count_alignment_error()
320                return empty_return
321
322        if self.a2 is None:
323            phreds, methylations, nucleotides = {}, {}, {}
324            for q_pos, r_pos in self.a.get_aligned_pairs(matches_only=True):
325                phreds[r_pos] = self.a.query_qualities[q_pos]
326                nucleotides[r_pos] = self.a.query_sequence[q_pos]
327                methylations[r_pos] = self._get_met_str(self.a)[q_pos]
328        else: # it's a paired alignment
329            if self.use_quality_profile:
330                from jtmethtools._quality_profiles import  quality_profile_match_41, quality_profile_mismatch_41
331            else:
332                quality_profile_match_41, quality_profile_mismatch_41 = None, None
333
334            # .aligned_pairs is reference position->read locus
335            a1_loc_pos, a2_loc_pos = [
336                {r: q for (q, r) in ap if (q is not None) and (r is not None)}
337                for ap in (self.a.get_aligned_pairs(), self.a2.get_aligned_pairs())
338            ]
339            # a1_loc_pos, a2_loc_pos = [
340            #     dict(aln.get_aligned_pairs(matches_only=True))
341            #     for aln in (self.a, self.a2)
342            # ]
343
344            a1_loc = set(a1_loc_pos.keys())
345            a2_loc = set(a2_loc_pos.keys())
346            shared_loc = a1_loc.intersection(a2_loc)
347            a1_only = a1_loc.difference(a2_loc)
348            a2_only = a2_loc.difference(a1_loc)
349
350            phreds = {}
351            nucleotides = {}
352            methylations = {}
353
354            a1_metstr = get_bismark_met_str(self.a)
355            a2_metstr = get_bismark_met_str(self.a2)
356
357            try:
358                # get the values where the position only exists in one of the mates
359                for only_loc, loc_pos, a, metstr in (
360                        (a1_only, a1_loc_pos, self.a, a1_metstr),
361                        (a2_only, a2_loc_pos, self.a2, a2_metstr)
362                ):
363                    for l in only_loc:
364                        p = loc_pos[l]
365                        phreds[l] = a.query_qualities[p]
366                        nucleotides[l] = a.query_sequence[p]
367                        methylations[l] = metstr[p]
368            except:
369                print('BAM = ', self.filename, ' | Read = ', self.a.query_name)
370                raise
371
372            if (not self.prefer_R1) or self.use_quality_profile:
373                for l in shared_loc:
374                    a1_pos = a1_loc_pos[l]
375                    a2_pos = a2_loc_pos[l]
376
377                    a1_nt = self.a.query_sequence[a1_pos]
378                    a2_nt = self.a2.query_sequence[a2_pos]
379
380                    a1_phred = self.a.query_qualities[a1_pos]
381                    a2_phred = self.a2.query_qualities[a2_pos]
382
383                    a1_met = a1_metstr[a1_pos]
384                    a2_met = a2_metstr[a2_pos]
385
386                    # Keep the nucleotide with the highest quality
387                    # (in the case that a1 and a2 have different nucleotides and
388                    #   the phred is the same, we'll keep the a1 NT)
389                    if a1_phred >= a2_phred:
390                        nt = a1_nt
391                        met = a1_met
392                    else:
393                        nt = a2_nt
394                        met = a2_met
395
396                    if self.use_quality_profile:
397                        if a1_nt == a2_nt:
398                            phred = quality_profile_match_41[a1_phred][a2_phred]
399                        else:
400                            phred = quality_profile_mismatch_41[a1_phred][a2_phred]
401                    else:
402                        phred = max((a1_phred, a2_phred))
403
404                    phreds[l] = phred
405                    nucleotides[l] = nt
406                    methylations[l] = met
407            else:
408                # use R1
409                for l in shared_loc:
410                    a1, loc_pos, metstr = (
411                        (self.a, a1_loc_pos, a1_metstr)
412                        if self.a.is_read1
413                        else (self.a2, a2_loc_pos, a2_metstr)
414                    )
415                    a1_pos = loc_pos[l]
416                    phreds[l] = a1.query_qualities[a1_pos]
417                    nucleotides[l] = a1.query_sequence[a1_pos]
418                    methylations[l] = a1_metstr[a1_pos]
419
420        sorted_phreds, sorted_nucleotides, sorted_methylations = (
421            dict(sorted(d.items()))
422            for d in (phreds, nucleotides, methylations)
423        )
424
425        return LocusValues(qualities=sorted_phreds, nucleotides=sorted_nucleotides, methylations=sorted_methylations)
